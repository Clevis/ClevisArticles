 44:         {

 46:         }

 48:             'service' => NULL,

 51:         $this->services[$name] = $tmp;
 52:         return $this;
 53:     }

 56:      * @param string
 57:      * @param string|NULL
 58:      * @return object
 59:      * @throws ServiceNotFoundException
 60:      * @throws InvalidServiceFactoryException
 61:      * @throws ServiceNotInstanceOfException if $instanceof not match with service
 62:      */
 63:     public function getService($name, $instanceof = NULL)
 64:     {
 65:         $this->hasService($name, true);
 66:         $s = $this->services[$name];
 67:         if ($s->service === NULL)
 68:         {
 69: 
 70:             if (is_string($s->factory) AND !(strpos($s->factory, '::') OR strncmp($s->factory, "\0lambda_", 8) === 0))
 71:             {
 72:                 $s->service = new $s->factory;
 73:                 unset($s->factory);
 74:             }
 75:             else if (Callback::is($s->factory) OR $s->factory instanceof Closure OR is_string($s->factory) OR is_array($s->factory))
 76:             {
 77:                 $tmp = Callback::create($s->factory)->invoke($this);
 78:                 if (!is_object($tmp))
 79:                 {
 80:                     $tmp = gettype($tmp);
 81:                     throw new InvalidServiceFactoryException("Factory for service '$name' returns invalid result. Object expected, $tmp given.");
 82:                 }
 83:                 $s->service = $tmp;
 84:                 unset($s->factory);
 85:             }
 86:             else if (is_object($s->factory))
 87:             {
 88:                 $s->service = $s->factory;
 89:                 unset($s->factory);
 90:             }
 91:             else
 92:             {
 93:                 $tmp = gettype($s->factory);
 94:                 throw new InvalidServiceFactoryException("Service '$name' has invalid factory. Callback, class name or object expected, $tmp given.");
 95:             }
 96:         }
 97:         if ($instanceof !== NULL AND !($s->service instanceof $instanceof))
 98:         {
 99:             throw new ServiceNotInstanceOfException("Service '$name' is not instance of '$instanceof'.");
100:         }
101:         return $s->service;
102:     }
