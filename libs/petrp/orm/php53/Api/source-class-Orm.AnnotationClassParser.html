 13: /**
 14:  * Loads class name in annotations.
 15:  *
 16:  * Annotation is only for some interface.
 17:  * It support inheritance.
 18:  *
 19:  * <code>
 20:  * $parser = new AnnotationClassParser;
 21:  * $parser->register('foo', 'FooInterface');
 22:  *
 23:  * /** @foo BarClass * /
 24:  * class FooClass implements FooInterface {}
 25:  *
 26:  * $parser->get('foo', new FooClass); // returns BarClass
 27:  * </code>
 28:  * @author Petr Procházka
 29:  * @package Orm
 30:  * @subpackage Mappers\Factory
 31:  */
 32: class AnnotationClassParser extends Object
 33: {
 34:     /** @var AnnotationsParser */
 35:     private $parser;
 36: 
 37:     /** @var array of name => stdClass */
 38:     private $registered = array();
 39: 
 40:     /** @param AnnotationsParser */
 41:     public function __construct(AnnotationsParser $parser)
 42:     {
 43:         $this->parser = $parser;
 44:     }
 45: 
 46:     /**
 47:      * <code>
 48:      *  $p->register('mapper', 'Orm\IRepository', function ($repositoryClass) {
 49:      *      return $repositoryClass . 'Mapper';
 50:      *  });
 51:      * </code>
 52:      *
 53:      * @param string
 54:      * @param string interface name
 55:      * @param Callback|Closure|NULL
 56:      * @return AnnotationClassParser
 57:      * @throws AnnotationClassParserException
 58:      */
 59:     public function register($annotation, $interface, $defaultClassFallback = NULL)
 60:     {
 61:         if (isset($this->registered[$annotation]))
 62:         {
 63:             throw new AnnotationClassParserException("Parser '$annotation' is already registered");
 64:         }
 65:         if (!interface_exists($interface))
 66:         {
 67:             throw new AnnotationClassParserException("'$interface' is not valid interface");
 68:         }
 69:         if ($defaultClassFallback !== NULL AND !is_callable($defaultClassFallback) AND !Callback::is($defaultClassFallback))
 70:         {
 71:             $tmp = is_string($defaultClassFallback) ? $defaultClassFallback : (is_object($defaultClassFallback) ? get_class($defaultClassFallback) : gettype($defaultClassFallback));
 72:             throw new AnnotationClassParserException("'$tmp' is not valid callback");
 73:         }
 74:         $tmp = (object) array(
 75:             'annotation' => $annotation,
 76:             'interface' => $interface,
 77:             'defaultClassFallback' => $defaultClassFallback,
 78:             'cache' => array(),
 79:         );
 80:         $this->registered[$annotation] = $tmp;
 81:         return $this;
 82:     }
 83: 
 84:     /**
 85:      * Get class name for object.
 86:      * @param string
 87:      * @param object
 88:      * @return string class name
 89:      * @throws AnnotationClassParserException
 90:      * @throws AnnotationClassParserNoClassFoundException
 91:      * @throws AnnotationClassParserMorePossibleClassesException
 92:      */
 93:     public function get($annotation, $object)
 94:     {
 95:         if (!isset($this->registered[$annotation]))
 96:         {
 97:             throw new AnnotationClassParserException("parser '$annotation' is not registered");
 98:         }
 99:         if (!is_object($object))
100:         {
101:             $tmp = gettype($object);
102:             throw new AnnotationClassParserException("expected object, $tmp given");
103:         }
104:         $r = $this->registered[$annotation];
105:         if (!($object instanceof $r->interface))
106:         {
107:             $tmp = get_class($object);
108:             throw new AnnotationClassParserException("'$tmp' is not instance of {$r->interface}");
109:         }
110: 
111:         $class = get_class($object);
112: 
113:         if (!isset($r->cache[$class]))
114:         {
115:             $result = $this->getByReflection(
116:                 $r,
117:                 new ReflectionClass($class),
118:                 $this->defaultClassFallback($r, $class)
119:             );
120:             if (!$result)
121:             {
122:                 throw new AnnotationClassParserNoClassFoundException("$class::@$annotation no class found");
123:             }
124:             $r->cache[$class] = $result;
125:         }
126:         return $r->cache[$class];
127:     }
128: 
129:     /**
130:      * @param stdClass
131:      * @param string
132:      * @return string|NULL
133:      */
134:     private function defaultClassFallback(stdClass $r, $class)
135:     {
136:         if ($r->defaultClassFallback)
137:         {
138:             $defaultClass = call_user_func($r->defaultClassFallback, $class);
139:             if (class_exists($defaultClass))
140:             {
141:                 return $defaultClass;
142:             }
143:         }
144:         return NULL;
145:     }
146: 
147:     /**
148:      * @param stdClass
149:      * @param string|false
150:      * @return string|false
151:      */
152:     private function getByClassName(stdClass $r, $class)
153:     {
154:         if (!$class)
155:         {
156:             return NULL;
157:         }
158:         if (!isset($r->cache[$class]))
159:         {
160:             $r->cache[$class] = false;
161:             $reflection = new ReflectionClass($class);
162:             $defaultClass = NULL;
163:             if ($reflection AND $reflection->implementsInterface($r->interface))
164:             {
165:                 if ($reflection->isInstantiable())
166:                 {
167:                     if ($dc = $this->defaultClassFallback($r, $class))
168:                     {
169:                         $dcReflection = new ReflectionClass($dc);
170:                         if ($dcReflection->isInstantiable())
171:                         {
172:                             $defaultClass = $dcReflection->getName();
173:                         }
174:                     }
175:                 }
176:                 $r->cache[$class] = $this->getByReflection($r, $reflection, $defaultClass);
177:             }
178:         }
179:         return $r->cache[$class];
180:     }
181: 
182:     /**
183:      * @param stdClass
184:      * @param string
185:      * @return string|false
186:      * @throws AnnotationClassParserException
187:      * @throws AnnotationClassParserMorePossibleClassesException
188:      */
189:     private function getByReflection(stdClass $r, ReflectionClass $reflection, $defaultClass)
190:     {
191:         $annotation = $this->parser->getByReflection($reflection);
192:         if (isset($annotation[$r->annotation]))
193:         {
194:             if (count($annotation[$r->annotation]) !== 1)
195:             {
196:                 throw new AnnotationClassParserException('Cannot redeclare ' . $reflection->getName() . '::@' . $r->annotation);
197:             }
198:             $class = $annotation[$r->annotation][0];
199:             if ($class === false)
200:             {
201:                 $defaultClass = NULL;
202:             }
203:             else
204:             {
205:                 if (!is_string($class))
206:                 {
207:                     $tmp = gettype($class);
208:                     throw new AnnotationClassParserException($reflection->getName() . "::@{$r->annotation} expected class name, $tmp given");
209:                 }
210:                 if (PHP_VERSION_ID >= 50300 AND ($ns = $reflection->getNamespaceName()) !== '' AND class_exists($ns . '\\' . $class))
211:                 {
212:                     $class = $ns . '\\' . $class;
213:                 }
214:                 else if (!class_exists($class))
215:                 {
216:                     throw new AnnotationClassParserException($reflection->getName() . "::@{$r->annotation} class '$class' not exists");
217:                 }
218:                 if ($defaultClass AND strcasecmp($class, $defaultClass) !== 0)
219:                 {
220:                     throw new AnnotationClassParserMorePossibleClassesException('Exists annotation ' . $reflection->getName() . '::@' . $r->annotation . " and fallback '$defaultClass'");
221:                 }
222:                 return $class;
223:             }
224:         }
225:         if ($defaultClass)
226:         {
227:             return $defaultClass;
228:         }
229: 
230:         return $this->getByClassName($r, get_parent_class($reflection->getName()));
231:     }
232: 
233:     /** @deprecated */
234:     final protected function getAnnotations(ReflectionClass $reflection)
235:     {
236:         throw new DeprecatedException(array(__CLASS__, 'getAnnotations()', '', __CLASS__ . '->parser->getByReflection()'));
237:     }
